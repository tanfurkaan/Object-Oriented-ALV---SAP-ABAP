  WHEN '&SAVE'.

ENDCASE.
ENDMODULE.

FORM get_total_sum .
DATA: lv_total_sum TYPE int4,
      lv_total_sum_c TYPE char10,
      lv_mess TYPE char200.

LOOP AT gt_list INTO gs_list.
lv_total_sum = lv_total_sum + gs_list-cost.
ENDLOOP.

lv_total_sum_c = lv_total_sum.

CONCATENATE 'Girilen hesapların toplamı = '
            lv_total_sum_c
            INTO lv_mess.

MESSAGE lv_mess TYPE 'I'.
ENDFORM.
